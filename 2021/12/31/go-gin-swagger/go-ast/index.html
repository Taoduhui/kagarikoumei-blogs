# Go AST Type

# AssignStmt 赋值、声明

type AssignStmt struct {
        Lhs    []Expr      // 等号左边
        TokPos token.Pos   // Tok的位置
        Tok    token.Token // 赋值令牌，DEFINE
        Rhs    []Expr      // 等号右边
}

# CallExpr 函数调用

type CallExpr struct {
        Fun      Expr      // 函数表达
        Lparen   token.Pos // “（”的位置
        Args     []Expr    // 函数参数; 或者没有
        Ellipsis token.Pos // “......”的位置（如果没有“......”则为token.NoPos）
        Rparen   token.Pos // “）的位置
}
